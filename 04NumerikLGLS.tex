\section{Numerik von LGLS}

\subsection{LR-Zerlegung}

\begin{displaymath}
	A = P \cdot L \cdot R
\end{displaymath}

\begin{itemize}
    \item Zeilentausch: $P$
    \item Gauss-Faktoren: $L$
    \item Stufenform: $R$
\end{itemize}

\subsection{QR-Zerlegung}

\begin{displaymath}
	A = Q \cdot R
\end{displaymath}

\begin{itemize}
    \item $Q \in O(n)$
    \item Stufenform: $R$
    \item $\det(Q) \in \{ -1; 1 \}$
    \item $spec(Q) \subseteq \{ -1; 1\}$
\end{itemize}

\begin{displaymath}
	\det(A) = \det(Q) \cdot \det(R) = \pm \det(R)
\end{displaymath}
